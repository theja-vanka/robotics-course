#
# Unpinned on purpose: pip resolves ONE latest, mutually-compatible set, which avoids the
# version-conflict whack-a-mole. Install cleanest into a FRESH env so nothing pre-installed
# clashes:
#     python -m venv .venv && source .venv/bin/activate
#     pip install -r requirements.txt
# (Installing into an existing env instead? add -U so pip upgrades to the latest compatible.)
# On a GPU box torch is usually preinstalled; pip keeps whatever satisfies the requirement.
# ─────────────────────────────────────────────────────────────────────────────

numpy==1.26.4

# ── Robot-scene frame reader (helpers/datasets.py) ─────────────────────────────
# The datasets store frames as AV1 MP4s. opencv-python is the primary decoder, but its
# bundled ffmpeg can't software-decode AV1 — so imageio-ffmpeg handles AV1 (no `av`/PyAV,
# no system ffmpeg needed).
opencv-python-headless   # cv2: image ops + decoding non-AV1 video
imageio                  # AV1 fallback reader ...
imageio-ffmpeg           # ... bundles an ffmpeg that decodes AV1
huggingface_hub          # downloads the dataset video clips (cached)

# ── Embeddings + vector DB ─────────────────────────────────────────────────────
transformers             # CLIP / DINOv2 encoders (also the depth-anything pipeline)
pymilvus[milvus_lite]    # file-based vector DB; the milvus_lite extra is REQUIRED for local .db

onnxruntime
